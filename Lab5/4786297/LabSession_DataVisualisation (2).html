# YOUR CODE HERE
data_year = df.query("year==2007")
pop = []
continents = data_year.continent.unique()
c = []
for continent in continents:
    pop.append(data_year[data_year.continent == continent]['pop'].sum())
    c.append(continent)

data_grouped = pd.DataFrame([continents,pop] ).transpose()
data_grouped.columns = ['continent', 'pop']
data_grouped = data_grouped.sort_values('pop', ascending = False)

fig = px.bar(
    data_grouped, 
    x = 'pop', 
    y = 'continent', 
    color = 'continent', 
    barmode = 'stack',
    text=pop,
    orientation = 'h',
)

fig.show()
